<script>
    (function() {
        // --- 配置区域 ---
        const basePdfUrl = '/papers/';          // PDF 存放的基础路径
        const jsonUrl = '/papers/list.json';     // 自动生成的 JSON 列表路径

        // --- DOM 元素 ---
        const fileListEl = document.getElementById('fileList');
        const currentLabel = document.getElementById('currentFileLabel');
        const pdfCanvas = document.getElementById('pdfCanvas');
        const pdfContainer = document.getElementById('pdf-container');
        const loadingEl = document.getElementById('loading');
        const prevBtn = document.getElementById('prevPage');
        const nextBtn = document.getElementById('nextPage');
        const pageNumSpan = document.getElementById('pageNum');
        const pageCountSpan = document.getElementById('pageCount');

        // --- 状态 ---
        let items = [];            // 从 JSON 获取的文件名列表（无扩展名）
        let currentItem = null;
        let pdfDoc = null;
        let currentPage = 1;
        let totalPages = 0;

        // --- 获取文件列表 ---
        async function fetchItems() {
            try {
                const res = await fetch(jsonUrl);
                if (!res.ok) throw new Error('网络错误');
                items = await res.json();
                if (items.length === 0) {
                    loadingEl.textContent = '暂无试卷';
                    return;
                }
                currentItem = items[0];
                renderSidebar();
                currentLabel.textContent = currentItem;
                loadPdf(currentItem);
            } catch (err) {
                console.error('列表加载失败', err);
                loadingEl.textContent = '❌ 加载列表失败';
            }
        }

        // --- 渲染侧边栏按钮 ---
        function renderSidebar() {
            fileListEl.innerHTML = '';
            items.forEach(item => {
                const btn = document.createElement('button');
                btn.className = 'file-btn' + (item === currentItem ? ' active' : '');
                btn.textContent = item;
                btn.addEventListener('click', () => {
                    if (item === currentItem) return;
                    document.querySelectorAll('.file-btn').forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    currentItem = item;
                    currentLabel.textContent = item;
                    loadPdf(item);
                });
                fileListEl.appendChild(btn);
            });
        }

        // --- 加载 PDF ---
        async function loadPdf(item) {
            loadingEl.style.display = 'block';
            pdfCanvas.style.display = 'none';
            prevBtn.disabled = true;
            nextBtn.disabled = true;
            pageNumSpan.textContent = '1';
            pageCountSpan.textContent = '?';

            const url = basePdfUrl + item + '.pdf';
            try {
                if (pdfDoc) pdfDoc.destroy();
                pdfDoc = await pdfjsLib.getDocument(url).promise;
                totalPages = pdfDoc.numPages;
                pageCountSpan.textContent = totalPages;
                currentPage = 1;
                await renderPage(currentPage);
                updateNavButtons();
                pdfCanvas.style.display = 'block';
                loadingEl.style.display = 'none';
            } catch (err) {
                console.error('PDF 加载失败', err);
                loadingEl.textContent = '❌ 加载失败';
            }
        }

        async function renderPage(pageNum) {
            const page = await pdfDoc.getPage(pageNum);
            const viewport = page.getViewport({ scale: 1.5 });
            const canvas = pdfCanvas;
            const ctx = canvas.getContext('2d');
            canvas.height = viewport.height;
            canvas.width = viewport.width;
            await page.render({ canvasContext: ctx, viewport }).promise;
            pageNumSpan.textContent = pageNum;
        }

        function updateNavButtons() {
            prevBtn.disabled = (currentPage <= 1);
            nextBtn.disabled = (currentPage >= totalPages);
        }

        prevBtn.addEventListener('click', async () => {
            if (currentPage > 1) {
                currentPage--;
                await renderPage(currentPage);
                updateNavButtons();
            }
        });
        nextBtn.addEventListener('click', async () => {
            if (currentPage < totalPages) {
                currentPage++;
                await renderPage(currentPage);
                updateNavButtons();
            }
        });

        // --- 初始化 ---
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', fetchItems);
        } else {
            fetchItems();
        }
    })();
</script>
